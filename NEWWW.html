        // Function to create the HTML for the gallery items

        function createGalleryItems() {

            const galleryList = document.getElementById('main-gallery');

            for (const key in artworks) {

                const artwork = artworks[key];

                listItem.className = 'artwork-item';

                listItem.setAttribute('data-artwork-id', key);


                // Check if the image source is an object with a 'gallery' property

                const imageSrc = (typeof imageSources[key] === 'object' && imageSources[key].gallery) 

                                 ? imageSources[key].gallery 

                                 : (imageSources[key] || 'https://placehold.co/500x500/A0A0A0/ffffff?text=Image+Not+Found');


                listItem.innerHTML = `

                    <img src="${imageSrc}" alt="${artwork.name}" class="artwork-image">

                    <div class="artwork-title">${artwork.name}</div>

                `;

                

                listItem.addEventListener('click', () => showArtworkDetail(key));
